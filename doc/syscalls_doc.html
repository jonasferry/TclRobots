DESCRIPTION

47 proc _ping_check {val} {
48     if {[lindex $val 0] eq "alert"} {
49         #  ping_proc:        dsp:
50         [lindex $val 1] [lindex $val 2]
51         set val [lrange $val 3 end]
52     }
53     return $val
54 }

   _syscall_yield

DESCRIPTION

69 proc _syscall_yield {args} {
70     set result [syscall callbackcheck]
71     if {$result ne ""} {
72         uplevel \#0 $result
73     }
74     syscall {*}$args
75     return [_ping_check [yield]]

NAME

   after

DESCRIPTION

   After does not work, emulate it using callbacks

SOURCE

749 proc after {ms args} {
750     set ticks [expr {$ms / 100}]
751     if {[llength $args] == 0} {
752         # Emulate a wait using some syscalls
753         for {set t 0} {$t < $ticks} {incr t} {
754             syscall speed
755             return [_ping_check [yield]]
756         }
757 
758     } else {
759         set script [join $args]
760         syscall callback $ticks $script
761     }
762     return
763 }

6.2. deprecated_cmds/damage [ procedure ]

NAME

   damage

DESCRIPTION

   damage

   The damage command reports the current percent of damage suffered by
   the robot, 0-99. At 100% damage, the robot is "dead", and as such,
   the control program is no longer running.
