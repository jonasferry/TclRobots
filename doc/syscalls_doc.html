NAME

   _ping_check

DESCRIPTION

   Check the answer from a yield for a ping alert

SOURCE

42 proc _ping_check {val} {
43     if {[lindex $val 0] eq "alert"} {
44         #  ping_proc:        dsp:
45         [lindex $val 1] [lindex $val 2]
46         set val [lrange $val 3 end]
47     }
48     return $val
49 }

2.2. helper/_syscall_yield [ Procedure ]

NAME

   _syscall_yield

DESCRIPTION

   Combine the common syscall+yield call

SOURCE

64 proc _syscall_yield {args} {
65     set result [syscall callbackcheck]
66     if {$result ne ""} {
67         uplevel \#0 $result
68     }
69     syscall {*}$args
70     return [_ping_check [yield]]
71 }

NAME

   after

DESCRIPTION

   After does not work, emulate it using callbacks

SOURCE

664 proc after {ms args} {
665     set ticks [expr {$ms / 100}]
666     if {[llength $args] == 0} {
667         # Emulate a wait using some syscalls
668         for {set t 0} {$t < $ticks} {incr t} {
669             syscall speed
670             return [_ping_check [yield]]
671         }
672 
673     } else {
674         set script [join $args]
675         syscall callback $ticks $script
676     }
677     return
678 }
